package com.attendance;

public class App {
    
    public double calculatePercentage(int totalClasses, int attendedClasses) {
        if (totalClasses <= 0) return 0.0;
        return ((double) attendedClasses / totalClasses) * 100;
    }

    public String checkEligibility(double percentage) {
        if (percentage >= 75.0) {
            return "ELIGIBLE";
        } else if (percentage >= 65.0) {
            return "WARNING: LOW ATTENDANCE";
        } else {
            return "INELIGIBLE";
        }
    }

    public static void main(String[] args) {
        App app = new App();
        double myAttendance = app.calculatePercentage(100, 78);
        System.out.println("Attendance: " + myAttendance + "%");
        System.out.println("Status: " + app.checkEligibility(myAttendance));
    }
}

package com.payroll;

public class App {
    public double calculateNetSalary(double basicSalary) {
        double da = basicSalary * 0.40;   // 40% Dearness Allowance
        double pf = basicSalary * 0.12;   // 12% Provident Fund
        
        double grossSalary = basicSalary + da;
        double tax = grossSalary * 0.10;  // 10% Tax on Gross
        
        return grossSalary - (pf + tax);
    }

    public static void main(String[] args) {
        App app = new App();
        double net = app.calculateNetSalary(50000);
        System.out.println("Net Salary for Basic 50000: " + net);
    }
}

}

package com.health;

public class App {
    
    public double calculateBMI(double weightKg, double heightM) {
        if (heightM <= 0) return 0;
        return weightKg / (heightM * heightM);
    }

    public String assessRisk(double bmi, int age, String lifestyle) {
        // Base risk from BMI
        int score = 0;
        if (bmi >= 30.0) score += 3;      // Obese
        else if (bmi >= 25.0) score += 1; // Overweight
        
        // Age factor
        if (age > 50) score += 2;
        
        // Lifestyle factor
        if (lifestyle.equalsIgnoreCase("sedentary")) score += 2;

        // Final Category
        if (score >= 5) return "HIGH RISK";
        if (score >= 2) return "MODERATE RISK";
        return "LOW RISK";
    }

    public static void main(String[] args) {
        App app = new App();
        double bmi = app.calculateBMI(85, 1.75);
        System.out.println("Result: " + app.assessRisk(bmi, 55, "sedentary"));
    }
}

package com.health;

import static org.junit.jupiter.api.Assertions.assertEquals;
import org.junit.jupiter.api.Test;

public class AppTest {
    App app = new App();

    @Test
    public void testHighRisk() {
        // Obese BMI (~31), Over 50, Sedentary = High Risk
        double bmi = app.calculateBMI(95, 1.75);
        assertEquals("HIGH RISK", app.assessRisk(bmi, 55, "sedentary"));
    }

    @Test
    public void testLowRisk() {
        // Normal BMI, Young, Active = Low Risk
        double bmi = app.calculateBMI(65, 1.75);
        assertEquals("LOW RISK", app.assessRisk(bmi, 25, "active"));
    }
}
